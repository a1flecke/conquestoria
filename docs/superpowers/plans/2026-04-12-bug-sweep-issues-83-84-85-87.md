## Task 1: Add real-world civ city-name rosters (#84, and primary cause of #83)

**Root cause:** `src/systems/civ-definitions.ts` only defines `cityNames` for fantasy civs (Prydain, Annuvin, Wakanda, Avalon, Lothlorien, Narnia, Atlantis). Every real-world civ (Egypt, Rome, Greece, Mongolia, Babylon, Zulu, China, Persia, England, Aztec, Japan, India, France, Germany, Gondor, Rohan, Russia, Ottoman, Shire, Isengard, Spain, Viking) has `cityNames` undefined, so `getNamingPoolForCiv` in `src/systems/city-name-system.ts:21-26` falls through to `DEFAULT_CITY_NAMES` which begins with `'Alexandria', 'Thebes', 'Memphis', 'Carthage', 'Athens', 'Sparta', 'Rome'…`. That is why Zulu cities show Egyptian/Greek names (#84) and France's first city is Alexandria (#83).

```typescript
import { describe, it, expect } from 'vitest';
import { createUnit, getMovementRange, resetUnitTurn, UNIT_DEFINITIONS } from '@/systems/unit-system';
import type { GameMap } from '@/core/types';

function plainsMap(radius = 5): GameMap {
  const tiles: GameMap['tiles'] = {};
  for (let q = -radius; q <= radius; q++) {
    for (let r = -radius; r <= radius; r++) {
      tiles[`${q},${r}`] = { coord: { q, r }, terrain: 'plains', owner: null, improvement: 'none', improvementTurnsLeft: 0 } as any;
    }
  }
  return { width: 2 * radius + 1, height: 2 * radius + 1, tiles, wrapsHorizontally: false } as GameMap;
}

describe('plains movement (#85)', () => {
  it('warrior with 2 movement reaches exactly 2 plains rings', () => {
    const warrior = createUnit('warrior', 'p1', { q: 0, r: 0 });
    expect(warrior.movementPointsLeft).toBe(2);
    const range = getMovementRange(warrior, plainsMap(), {}, {});
    // 2-ring hex disc minus the origin tile = 6 + 12 = 18 hexes reachable
    expect(range.length).toBe(18);
  });

  it('non-Viking civs do NOT get a movement bonus', () => {
    const warrior = createUnit('warrior', 'p1', { q: 0, r: 0 }, { type: 'auto_roads' });
    expect(warrior.movementPointsLeft).toBe(UNIT_DEFINITIONS.warrior.movementPoints);
    expect(warrior.movementBonus).toBeUndefined();
  });

  it('resetUnitTurn does not accumulate bonus', () => {
    let warrior = createUnit('warrior', 'p1', { q: 0, r: 0 });
    warrior = resetUnitTurn(warrior);
    warrior = resetUnitTurn(warrior);
    expect(warrior.movementPointsLeft).toBe(2);
  });

  it('Viking warrior reaches exactly 3 plains rings', () => {
    const warrior = createUnit('warrior', 'p1', { q: 0, r: 0 }, { type: 'naval_raiding', movementBonus: 1, coastalVisionBonus: 1 });
    expect(warrior.movementPointsLeft).toBe(3);
    const range = getMovementRange(warrior, plainsMap(), {}, {});
    // 3-ring hex disc = 6 + 12 + 18 = 36
    expect(range.length).toBe(36);
  });
});
```
